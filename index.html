from flask import Flask, render_template, request import os import re app = Flask(__name__) UPLOAD_FOLDER = "uploads" app.config["UPLOAD_FOLDER"] = UPLOAD_FOLDER # Create uploads folder automatically os.makedirs(UPLOAD_FOLDER, exist_ok=True) # Home Page @app.route("/") def home(): return render_template("index.html") # URL Detection @app.route("/url", methods=["POST"]) def url_detection(): user_url = request.form["url"].lower() suspicious_keywords = [ "login", "verify", "secure", "bank", "free", "bonus", "gift", "win", "prize", "claim", "password", "update", "wallet", "crypto", "lottery" ] suspicious_domains = [ ".xyz", ".tk", ".ru", ".cn" ] score = 0 # Check suspicious keywords for word in suspicious_keywords: if word in user_url: score += 1 # Check suspicious domains for domain in suspicious_domains: if domain in user_url: score += 1 # Detect IP-based URLs ip_pattern = r"(https?:\/\/)?(\d{1,3}\.){3}\d{1,3}" if re.search(ip_pattern, user_url): score += 2 # Long URLs can be suspicious if len(user_url) > 75: score += 1 # Final result if score >= 2: result = "Malicious" else: result = "Benign" return render_template( "index.html", prediction=result, entered_url=user_url ) # File Detection @app.route("/scan", methods=["POST"]) def scan_file(): uploaded_file = request.files["file"] if uploaded_file.filename == "": return render_template( "index.html", file_result="No File Selected" ) file_path = os.path.join( app.config["UPLOAD_FOLDER"], uploaded_file.filename ) uploaded_file.save(file_path) suspicious_text = [ "win money", "free recharge", "claim prize", "urgent", "lottery", "click here", "verify account", "bank update", "free gift", "password reset" ] result = "Safe File" # Only scan txt files if uploaded_file.filename.endswith(".txt"): with open( file_path, "r", encoding="utf-8", errors="ignore" ) as f: content = f.read().lower() score = 0 for word in suspicious_text: if word in content: score += 1 if score >= 2: result = "Malicious File" return render_template( "index.html", file_result=result ) if __name__ == "__main__": app.run(debug=True)
